async function handleWynncraftCallback() { const urlParams = new URLSearchParams(window.location.search); const code = urlParams.get('code'); const state = urlParams.get('state'); const savedState = sessionStorage.getItem('wynn_oauth_state'); const codeVerifier = sessionStorage.getItem('wynn_code_verifier'); // 1. Verify CSRF state matches if (!state || state !== savedState) { console.error("State mismatch. Request might have been intercepted!"); return; } if (!code || !codeVerifier) { console.error("Authorization code or local code verifier is missing."); return; } // Clean up storage immediately sessionStorage.removeItem('wynn_code_verifier'); sessionStorage.removeItem('wynn_oauth_state'); // 2. Send authorization_code and the raw verifier to Wynncraft's token endpoint try { const response = await fetch('https://api.wynncraft.com/v3/oauth/token', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: new URLSearchParams({ grant_type: 'authorization_code', client_id: 'njUEJhhqs_eOJK73FKy-f3DmBpCm-Z48', code: code, redirect_uri: 'https://wanets.me/redirectservice', code_verifier: codeVerifier // Wynncraft's token exchange expects 'code_verifier' }) }); if (!response.ok) { throw new Error(`Token request failed: ${response.statusText}`); } const data = await response.json(); console.log("Token retrieved successfully!", data); // Store the bearer access_token securely localStorage.setItem('wynn_access_token', data.access_token); // Send back to your home page window.location.href = '/'; } catch (error) { console.error("Failed to complete OAuth token exchange:", error); } } // Automatically runs on redirect landing if (window.location.pathname.includes('/redirectservice')) { handleWynncraftCallback(); }
